SELECT ISSUE.NAME, I.SUMMARY
FROM ISSUESLISTS ISSUE
JOIN ISSUES I ON ISSUE.ID = I.ISSUESLIST_ID;

SELECT U.LASTNAME, U.FIRSTNAME, I.SUMMARY, I.DESCRIPTION
FROM USERS U
JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO;

SELECT U.LASTNAME, U.FIRSTNAME, COUNT(*) AS TASK_NUM
FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY I.USER_ID_ASSIGNEDTO
ORDER BY U.LASTNAME;
